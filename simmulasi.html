local bullet = script.Parent local debris = game:GetService("Debris") -- Tentukan batas area permainan (sesuaikan dengan ukuran map kamu) local BATAS_KIRI_X = -50 -- Contoh koordinat X paling kiri local BATAS_KANAN_X = 50 -- Contoh koordinat X paling kanan local damageAktif = true local damageAmount = 20 -- 1. FUNGSI DAMAGE DAN DETEKSI COUNTER bullet.Touched:Connect(function(otherPart) -- Cek apakah yang menyentuh adalah peluru pemain (untuk counter) -- *Pastikan peluru pemain kamu diberi nama "PlayerBullet" atau memiliki Tag khusus* if otherPart.Name == "PlayerBullet" then -- Efek visual hancur (opsional) local explosion = Instance.new("Explosion") explosion.Position = bullet.Position explosion.Parent = game.Workspace -- Hancurkan kedua peluru bullet:Destroy() otherPart:Destroy() return end -- Cek apakah yang menyentuh adalah Karakter Pemain (Humanoid) local humanoid = otherPart.Parent:FindFirstChildOfClass("Humanoid") if humanoid and damageAktif then -- Pastikan tidak melukai bos itu sendiri (misal bos diberi nama "Boss") if otherPart.Parent.Name ~= "Boss" then humanoid:TakeDamage(damageAmount) bullet:Destroy() -- Hancurkan peluru setelah kena pemain end end end) -- 2. LOOP UNTUK MENGECEK POSISI PELURU (BATAS LAYAR KANAN/KIRI) -- Menggunakan Task.spawn agar loop berjalan tanpa menghentikan script lain task.spawn(function() while bullet and bullet.Parent do local posisiX = bullet.Position.X -- Jika peluru keluar dari batas kanan atau kiri if posisiX < BATAS_KIRI_X or posisiX > BATAS_KANAN_X then damageAktif = false -- Matikan damage agar tidak terhitung -- Opsional: Langsung hancurkan peluru agar tidak memenuhi memory (lag) bullet:Destroy() break end task.wait(0.1) -- Cek setiap 0.1 detik untuk menghemat performa end end) -- Kebijakan keamanan: Hancurkan peluru otomatis setelah 5 detik jika tidak menyentuh apapun debris:AddItem(bullet, 5)
